62          final Type superclass = subclass.getGenericSuperclass();
63          if (superclass instanceof Class<?>) {

69      @Override public final Object invoke(final Object proxy, final Method method, final Object[] args) throws Throwable {
70          methodInvokationContext.set(new MethodInvokationContext(proxy, method, args));
71          try {
72              for (final Entry<Matcher<? super ReflectedMethod>, MethodBody> registeredMethodHandler : registeredMethodHandlers
73                      .entrySet()) {
74                  if (registeredMethodHandler.getKey().matches(methodInvokationContext.get().method)) {
75                      final MethodBody registeredImplementation = registeredMethodHandler.getValue();
76  
77                      registeredImplementation.body();
78  
79                      return methodInvokationContext.get().result;
80                  }
81              }
82              throw new UnsupportedOperationException("no implemention found for method " + method);
83          } finally {
84              methodInvokationContext.set(null);
85          }
86      }

92      public final MethodBinding<T> matching(final Matcher<? super ReflectedMethod> methodMatcher) {
93          return new MethodBinding<T>() {
94              @Override public void execute(final MethodBody methodBody) {
95                  registeredMethodHandlers.put(methodMatcher, methodBody);
96              }

98              @Override public void execute(final QueryMethod queryMethod) {
99                  execute(new MethodBody() {
100                     @Override public void body() throws Exception {
101                         final Method method = queryMethod.getClass().getDeclaredMethods()[0];
102                         try {
103                             method.setAccessible(true);
104                             returnValue(method.invoke(queryMethod, args()));
105                         } catch (final SecurityException e) {
106                             throw new SecurityException("unable to invoke method in " + queryMethod.getClass(), e);
107                         } catch (final IllegalAccessException e) {
108                             throw new IllegalAccessRuntimeException("unable to invoke method in "
109                                     + queryMethod.getClass(), e);
110                         } catch (final InvocationTargetException e) {
111                             throw new InvocationTargetRuntimeException(e, method);
112                         }
113                     }
114                 });
115             }
116         };
