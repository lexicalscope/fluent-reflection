37  class ReflectedMethodImpl extends AbstractReflectedCallable implements ReflectedMethod {
38      private final ReflectedTypeFactory reflectedTypeFactory;
39      private final ReflectedClass<?> reflectedClass;
40      private final TypeLiteral<?> typeLiteral;
41      private final Method method;

43      public ReflectedMethodImpl(final ReflectedTypeFactory reflectedTypeFactory,
44              final ReflectedClass<?> reflectedClass,
45              final TypeLiteral<?> typeLiteral,
46              final Method method) {
47          super(reflectedTypeFactory, method);
48          this.reflectedTypeFactory = reflectedTypeFactory;
49          this.reflectedClass = reflectedClass;
50          this.typeLiteral = typeLiteral;
51          this.method = method;
52      }

54      @Override public String getName() {
55          return method.getName();
56      }

58      @Override public List<ReflectedClass<?>> argumentTypes() {
59          final List<ReflectedClass<?>> result = new ArrayList<ReflectedClass<?>>();
60          result.addAll(convert(
61                  typeLiteral.getParameterTypes(method),
62                  new ConvertTypeLiteralToReflectedType(reflectedTypeFactory)));
63          return result;
64      }

66      @Override public int argumentCount() {
67          final int parameterCount = method.getParameterTypes().length;
68          if (isStatic()) {
69              return parameterCount;
70          }
71          return parameterCount;
72      }
73  
74      @Override public ReflectedClass<?> declaringClass() {
75          return reflectedTypeFactory.reflect(typeLiteral);
76      }
77  
78      @Override public Object call(final Object... args) {
79          if (isStatic()) {
80              return invokeMethod(null, args);
81          } else {
82              if (args.length < 1) {
83                  throw new IllegalArgumentException("target instance must be specified as first argument when calling "
84                          + method);
85              }
86  
87              final Object[] remainingArguments = new Object[args.length - 1];
88              arraycopy(args, 1, remainingArguments, 0, args.length - 1);
89              return invokeMethod(args[0], remainingArguments);
90          }
91      }
92  
93      private Object invokeMethod(final Object instance, final Object[] arguments) {
94          try {
95              if (!method.isAccessible()) {
96                  method.setAccessible(true);
97              }
98              return method.invoke(instance, arguments);
99          } catch (final IllegalArgumentException e) {
100             throw new IllegalArgumentRuntimeException(e, method, instance, arguments);
101         } catch (final IllegalAccessException e) {
102             throw new IllegalAccessRuntimeException(e, method);
103         } catch (final InvocationTargetException e) {
104             throw new InvocationTargetRuntimeException(e, method);
105         }
106     }

108     @Override public boolean isStatic() {
109         return Modifier.isStatic(method.getModifiers());
110     }

112     @Override public boolean isFinal() {
113         return Modifier.isFinal(method.getModifiers());

116     @Override public <T> ReflectedQuery<T> castResultTo(final Class<T> returnType) {
117         return new ReflectedQuery<T>() {
118             @Override public T call(final Object... args) {
119                 return returnType.cast(ReflectedMethodImpl.this.call(args));
120             }
121         };
122     }

124     @Override public ReflectedClass<?> returnType() {
125         final TypeLiteral<?> returnType = typeLiteral.getReturnType(method);
126         if (returnType == null) {
127             return null;
128         }
129         return reflectedTypeFactory.reflect(returnType);
130     }

132     @Override public String propertyName() {
133         final String name = getName();
134 
135         if (name.length() > 2) {
136             if (name.length() > 3) {
137                 if (name.startsWith("get") || name.startsWith("set")) {
138                     return initialLowerCase(name.substring(3));
139                 }
140             }
141             if (name.startsWith("is")) {
142                 return initialLowerCase(name.substring(2));
143             }
144         }
145         return method.getName();
146     }

148     private String initialLowerCase(final String substring) {
149         return substring.substring(0, 1).toLowerCase() + substring.substring(1);
150     }

165     @Override public String toString() {
166         final String visibility;
167         if (visibility().toString().isEmpty())
168         {
169             visibility = visibility().toString();
170         }
171         else
172         {
173             visibility = visibility().toString() + " ";
174         }
175 
176         final String staticModifier;
177         if (isStatic())
178         {
179             staticModifier = "static ";
180         }
181         else
182         {
183             staticModifier = "";
184         }
185 
186         final String finalModifier;
187         if (isFinal())
188         {
189             finalModifier = "final ";
190         }
191         else
192         {
193             finalModifier = "";
194         }
195 
196         final String typeParameters;
197         if (typeParameters().isEmpty())
198         {
199             typeParameters = "";
200         }
201         else
202         {
203             typeParameters = "<" + joinFrom(typeParameters(), ", ").toString() + "> ";
204         }
205 
206         final String arguments;
207         if (argumentCount() > 0) {
208             arguments = joinFrom(argumentTypes(), ", ").toString();
209         } else {
210             arguments = "";
211         }
212 
213         return format(
214                 "%s%s%s%s%s %s(%s)",
215                 visibility,
216                 staticModifier,
217                 finalModifier,
218                 typeParameters,
219                 returnType(),
220                 method.getName(),
221                 arguments);
222     }

224     @Override public boolean equals(final Object obj) {
225         if (obj != null && this.getClass().equals(obj.getClass())) {
226             final ReflectedMethodImpl that = (ReflectedMethodImpl) obj;
227             return new EqualsBuilder()
228                     .append(this.method, that.method)
229                     .append(this.typeLiteral, that.typeLiteral)
230                     .isEquals();
231         }
232         return false;
233     }

235     @Override public int hashCode() {
236         return new HashCodeBuilder().append(method).append(typeLiteral).toHashCode();
237     }
