27  public class MapBean {
28      public static <T> T bean(final Class<T> klass, final Map<String, Object> map) {
29          return dynamicProxy(new Implementing<T>(klass) {
30              {
31                  matching(isGetter()).execute(new MethodBody() {
32                      @Override public void body() {
33                          returnValue(map.get(method().propertyName()));
34                      }
35                  });
36  
37                  matching(isSetter()).execute(new MethodBody() {
38                      @Override public void body() {
39                          map.put(method().propertyName(), args()[0]);
40                      }
41                  });
42  
43                  matching(isExistence()).execute(new MethodBody() {
44                      @Override public void body() {
45                          returnValue(map.containsKey(method().propertyName()));
46                      }
47                  });
48              }
49          });
50      }
