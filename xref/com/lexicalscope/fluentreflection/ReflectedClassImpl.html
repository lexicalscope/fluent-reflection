35  /**
36   * Not thread safe
37   * 
38   * @author tim
39   * 
40   * @param <T>
41   */
42  class ReflectedClassImpl<T> implements ReflectedClass<T> {
43      private final ReflectedTypeFactory reflectedTypeFactory;
44      private final Class<T> klass;
45      private final ReflectedMembers<T> members;
46      private final TypeLiteral<T> typeLiteral;
47      private final ReflectedAnnotatedImpl annotatedElement;
48  
49      ReflectedClassImpl(
50              final ReflectedTypeFactory reflectedTypeFactory,
51              final TypeLiteral<T> typeLiteral,
52              final ReflectedMembers<T> members) {
53          this.reflectedTypeFactory = reflectedTypeFactory;
54          this.klass = (Class<T>) typeLiteral.getRawType();
55          this.typeLiteral = typeLiteral;
56          this.members = members;
57          this.annotatedElement = new ReflectedAnnotatedImpl(reflectedTypeFactory, klass);
58      }
59  
60      @Override public Class<T> classUnderReflection() {
61          return klass;
62      }
63  
64      @Override public List<ReflectedMethod> methods(final Matcher<? super ReflectedMethod> methodMatcher) {
65          return members.methods(methodMatcher);
66      }
67  
68      @Override public ReflectedMethod method(final Matcher<? super ReflectedMethod> methodMatcher) {
69          return members.method(methodMatcher);
70      }
71  
72      @Override public ReflectedMethod method(final String name) {
73          return members.method(callableHasName(name));
74      }
75  
76      @Override public ReflectedMethod staticMethod(final Matcher<? super ReflectedMethod> methodMatcher) {
77          return method(ReflectionMatchers.methodIsStatic().and(methodMatcher));
78      }
79  
80      @Override public List<ReflectedMethod> methods() {
81          return members.methods();
82      }
83  
84      @Override public List<ReflectedMethod> declaredMethods() {
85          return members.declaredMethods();
86      }
87  
88      @Override public List<ReflectedClass<?>> interfaces() {
89          return members.superclassesAndInterfaces(typeIsInterface());
90      }
91  
92      @Override public List<ReflectedClass<?>> superclasses() {
93          return members.superclassesAndInterfaces(not(typeIsInterface()));
94      }
95  
96      @Override public ReflectedClass<?> asType(final Matcher<ReflectedClass<?>> typeMatcher) {
97          if (typeMatcher.matches(this)) {
98              return this;
99          }
100         return Lambda.selectFirst(members.superclassesAndInterfaces(), typeMatcher);
101     }
102 
103     @Override public boolean isType(final ReflectionMatcher<ReflectedClass<?>> typeMatcher) {
104         if (typeMatcher.matches(this)) {
105             return true;
106         }
107         return hasItem(typeMatcher).matches(members.superclassesAndInterfaces());
108     }
109 
110     @Override public boolean isInterface() {
111         return typeLiteral.getRawType().isInterface();
112     }
113 
114     @Override public T constructRaw(final Object... args) {
115         final ReflectedConstructor<T> constructor =
116                 constructor(callableHasArgumentList(convert(args, new ConvertObjectToClass())));
117 
118         if (constructor == null) {
119             throw new ConstructorNotFoundRuntimeException(typeLiteral.getRawType());
120         }
121 
122         return constructor.call(args);
123     }
124 
125     @Override public ReflectedObject<T> construct(final Object... args) {
126         final T newInstance = constructRaw(args);
127         return reflectedTypeFactory.reflect(typeLiteral, newInstance);
128     }
129 
130     @Override public List<ReflectedConstructor<T>> constructors(
131             final Matcher<? super ReflectedConstructor<?>> constructorMatcher) {
132         return members.constructors(constructorMatcher);
133     }
134 
135     @Override public ReflectedConstructor<T> constructor(
136             final Matcher<? super ReflectedConstructor<?>> constructorMatcher) {
137         return members.constructor(constructorMatcher);
138     }
139 
140     @Override public boolean assignableFromObject(final Object value) {
141         return value == null
142                 || klass.isAssignableFrom(value.getClass())
143                 || canBeBoxed(value.getClass())
144                 || canBeUnboxed(value.getClass());
145     }
146 
147     @Override public boolean assignableTo(final Class<?> otherKlass) {
148         return otherKlass.isAssignableFrom(klass);
149     }
150 
151     @Override public boolean canBeBoxed(final Class<?> from) {
152         return Primitives.isWrapperType(klass)
153                 && Primitives.unwrap(klass).isAssignableFrom(from);
154     }
155 
156     @Override public boolean canBeUnboxed(final Class<?> from) {
157         return isPrimitive()
158                 && Primitives.wrap(klass).isAssignableFrom(from);
159     }
160 
161     @Override public ReflectedClass<T> boxedType() {
162         if (isPrimitive()) {
163             return reflectedTypeFactory.reflect(Primitives.wrap(klass));
164         }
165         return this;
166     }
167 
168     @Override public boolean isPrimitive() {
169         return klass.isPrimitive();
170     }
171 
172     @Override public ReflectedClass<?> typeArgument(final int typeParameter) {
173         return reflectedTypeFactory.reflect(TypeLiteral.get(((ParameterizedType) typeLiteral.getType())
174                 .getActualTypeArguments()[typeParameter]));
175     }
176 
177     @Override public String name() {
178         return klass.getName();
179     }
180 
181     @Override public String simpleName() {
182         return klass.getSimpleName();
183     }
184 
185     @Override public ReflectedClass<?> annotation(final Matcher<? super ReflectedClass<?>> annotationMatcher) {
186         return annotatedElement.annotation(annotationMatcher);
187     }
188 
189     @Override public boolean annotatedWith(final Class<? extends Annotation> annotationClass) {
190         return annotatedElement.annotatedWith(annotationClass);
191     }
192 
193     @Override public <A extends Annotation> A annotation(final Class<A> annotationClass) {
194         return annotatedElement.annotation(annotationClass);
195     }
196 
197     @Override public Type type() {
198         return typeLiteral.getType();
199     }
200 
201     @Override public boolean equals(final Object that) {
202         if (that != null && that.getClass().equals(this.getClass())) {
203             return typeLiteral.equals(((ReflectedClassImpl<?>) that).typeLiteral);
204         }
205         return false;
206     }
207 
208     @Override public int hashCode() {
209         return typeLiteral.hashCode();
210     }
211 
212     @Override public String toString() {
213         return typeLiteral.toString();
214     }
