14  final class MatcherMethodWithArguments extends ReflectionMatcher<ReflectedMethod> {
15      private final Class<?>[] expectedArgumentTypes;
16  
17      MatcherMethodWithArguments(Class<?>[] expectedArgumentTypes) {
18          this.expectedArgumentTypes = expectedArgumentTypes;
19      }
20  
21      @Override
22      public boolean matchesSafely(final ReflectedMethod arg) {
23          final List<Class<?>> actualArgumentTypes =
24                  convert(arg.argumentTypes(), new ConvertReflectedTypeToClass());
25  
26          if (expectedArgumentTypes == null || expectedArgumentTypes.length == 0) {
27              return actualArgumentTypes.size() == 0;
28          }
29          if (expectedArgumentTypes.length != actualArgumentTypes.size()) {
30              return false;
31          }
32          return arrayContaining(actualArgumentTypes.toArray(new Class[actualArgumentTypes.size()])).matches(
33                  expectedArgumentTypes);
34      }
35  
36      @Override
37      public void describeTo(final Description description) {
38          description.appendText("method with arguments ").appendValue(expectedArgumentTypes);
39      }
