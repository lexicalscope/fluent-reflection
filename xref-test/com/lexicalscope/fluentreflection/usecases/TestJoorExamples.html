29       *                 .call("substring", 6)  // Call the most specific matching substring() method
30       *                 .call("toString")      // Call toString()

44       * Employee[] employees = on(department).call("getEmployees").get();
45       * for (Employee employee : employees) {

47       *     System.out.println(street);
48       * }

50      @Test @SuppressWarnings("unused") public void joorArrayExample() {
51          class Address {
52              private final String street;
53              public Address(final String street) { this.street = street; }
54              String getStreet() { return street; }
55          }
56          class Employee {
57              private final String street;
58              public Employee(final String street) { this.street = street; }
59              Address getAddress() { return new Address(street); }
60          }
61          class Department {
62              Employee[] getEmployees() { return new Employee[]{new Employee("street0"), new Employee("street1")}; }
63          }
64  
65          final Employee[] employees = object(new Department()).call("getEmployees").as(Employee[].class);
66          for (int i = 0; i < employees.length; i++) {
67              assertThat(
68                      object(employees[i]).call("getAddress").call("getStreet").as(String.class),
69                      equalTo("street" + i));
70          }
71      }
